  2. Funci贸n que determine si un n煤mero es primo
  3. Funci贸n que convierta kil贸metros a millas (1 km = 0.621371 millas)

# 1
area_rectangulo <- function(base, altura) {
  return(base * altura)
}
area_rectangulo(5, 3)  # 15

# 2
es_primo <- function(n) {
  if (n <= 1) return(FALSE)
  if (n == 2) return(TRUE)
  for (i in 2:sqrt(n)) {
    if (n %% i == 0) return(FALSE)
  }
  return(TRUE)
}
es_primo(17)  # TRUE
es_primo(15)  # FALSE

# 3
km_a_millas <- function(km) {
  return(km * 0.621371)
}
km_a_millas(10)  # 6.21371

# 4
factorial_num <- function(n) {
  if (n == 0 || n == 1) return(1)
  resultado <- 1
  for (i in 2:n) {
    resultado <- resultado * i
  }
  return(resultado)
}
factorial_num(5)  # 120

# 5
mayor_de_tres <- function(a, b, c) {
  return(max(a, b, c))
}
mayor_de_tres(45, 78, 62)  # 78
